
CREATE TABLE `lims_project` (
  `project_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `original_id` varchar(100) DEFAULT NULL COMMENT '原系统ID',
  
  -- 项目核心信息
  `program_code` varchar(100) NOT NULL COMMENT '方案代码',
  `profile_type` varchar(100) DEFAULT NULL COMMENT '方案类型',
  `profile` varchar(500) DEFAULT NULL COMMENT '方案名称',
  `main_pro_num` varchar(100) DEFAULT NULL COMMENT '主项目数',
  
  -- 时间信息
  `start_use_date` datetime DEFAULT NULL COMMENT '启用日期',
  `expiration_date` datetime DEFAULT NULL COMMENT '到期日期',
  
  -- 状态信息
  `status` char(1) DEFAULT '0' COMMENT '状态（0正常 1停用）',
  `disp_status` varchar(50) DEFAULT NULL COMMENT '显示状态',
  `row_audit` varchar(50) DEFAULT NULL COMMENT '行审核状态',
  
  -- 价格信息
  `m_price` decimal(15,2) DEFAULT NULL COMMENT '价格',
  
  -- 说明信息
  `explain_desc` text COMMENT '方案说明',
  
  -- 若依标准字段
  `create_by` varchar(64) DEFAULT '' COMMENT '创建者',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_by` varchar(64) DEFAULT '' COMMENT '更新者',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `del_flag` char(1) DEFAULT '0' COMMENT '删除标志（0代表存在 1代表删除）',
  
  PRIMARY KEY (`project_id`),
  UNIQUE KEY `idx_program_code` (`program_code`),
  UNIQUE KEY `idx_original_id` (`original_id`),
  KEY `idx_profile_type` (`profile_type`),
  KEY `idx_status` (`status`),
  KEY `idx_disp_status` (`disp_status`),
  KEY `idx_start_use_date` (`start_use_date`),
  KEY `idx_expiration_date` (`expiration_date`),
  KEY `idx_main_pro_num` (`main_pro_num`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='方案表';
